592  void trim_deep(Figs_type &figs, Figc_type &figc) const
593  {
594  std::vector<size_type> elem;
595 
596  for (size_type i = 0; i < figs.size(); ++i)
597  {
598  int lcount = 0, rcount = 0;
599 
600  for (auto itr = figs[i].rbegin(); itr != figs[i].rend(); ++itr)
601  {

603  ++lcount;

609  {
610  if (*itr == ' ')
611  ++rcount;
612  else
613  break;
614  }
615 
616  elem.push_back(lcount + rcount);
617  }
618 
619  size_type space = *std::min_element(elem.begin(), elem.end());
620 
621  for (size_type i = 0; i < figs.size(); ++i)
622  {
623  size_type siz = space;
624 
625  while (siz > 0 && figs[i].back() == ' ')
626  {
627  figs[i].pop_back();
628  --siz;
629  }
630 
631  figc[i].erase(0, siz);
632  }
633  }

732  char_type smush_rules(char_type lc, char_type rc) const
733  {
734  //()
735  if (lc == ' ')
736  {
737  return rc;
738  }
739 
740  if (rc == ' ')
741  {
742  return lc;
743  }
744 
745  //(Equal character smush)
746  if (lc == rc)
747  {
748  return rc;
749  }
750 
751  //(Underscores smush)
752  if (lc == '_' && this->cvt("|/\\[]{}()<>").find(rc) != string_type::npos)
753  {
754  return rc;
755  }
756 
757  if (rc == '_' && this->cvt("|/\\[]{}()<>").find(lc) != string_type::npos)
758  {
759  return lc;
760  }
761 
762  //(Hierarchy Smushing)
763  auto find_class = [](char_type ch) -> size_type {
764  if (ch == '|')
765  {
766  return 1;
767  }
768 
769  if (ch == '/' || ch == '\\')
770  {
771  return 3;
772  }
773 
774  if (ch == '[' || ch == ']')
775  {
776  return 4;
777  }
778 
779  if (ch == '{' || ch == '}')
780  {
781  return 5;
782  }
783 
784  if (ch == '(' || ch == ')')
785  {
786  return 6;
787  }
788 
789  return 0;
790  };
791 
792  size_type c_lc = find_class(lc);
793  size_type c_rc = find_class(rc);
794 
795  if (c_lc > c_rc)
796  {
797  return lc;
798  }
799 
800  if (c_rc > c_lc)
801  {
802  return rc;
803  }
804 
805  //(Opposite smush)
806  if (lc == '[' && rc == ']')
807  {
808  return '|';
809  }
810 
811  if (lc == ']' && rc == '[')
812  {
813  return '|';
814  }
815 
816  if (lc == '{' && rc == '}')
817  {
818  return '|';
819  }
820 
821  if (lc == '}' && rc == '{')
822  {
823  return '|';
824  }
825 
826  if (lc == '(' && rc == ')')
827  {
828  return '|';
829  }
830 
831  if (lc == ')' && rc == '(')
832  {
833  return '|';
834  }
835 
836  //(Big X smush)
837  if (lc == '/' && rc == '\\')
838  {
839  return '|';
840  }
841 
842  if (lc == '\\' && rc == '/')
843  {
844  return 'Y';
845  }
846 
847  if (lc == '>' && rc == '<')
848  {
849  return 'X';
850  }
851 
852  //(universel smush)
853  return lc;
854  }
