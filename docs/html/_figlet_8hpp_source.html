290 void start(std::basic_istream<char_type, traits_type> &stream)
291 {
292 string_type str;
293 size_type dummy;
294 size_type cmnt;
295 int old_tag;
296
297 stream >> str;
298
299 if (str.substr(0, 5) != cvt("flf2a"))
300 {
301 throw std::runtime_error("flf2a not found !");
302 }
303
304 str.erase(0, 5);
305
306 this->HardBlank = str.at(0); // HardBlank
307
308 stream >> this->Height; // Height
309 stream >> dummy;
310 stream >> dummy;
311 stream >> old_tag;
312 stream >> cmnt; // comment line
313
314 std::getline(stream, str);
315
316 for (size_type i = 0; i < cmnt; i++)
317 {
318 std::getline(stream, str);
319 }
320
321 if (old_tag < 0)
322 {
323 this->Shrink = 0;
324 }
325 else if (old_tag == 0)
326 {
327 this->Shrink = 1;
328 }
329 else
330 {
331 this->Shrink = 2;

341 Figc_type next(std::basic_istream<char_type, traits_type> &stream)
342 {
343 std::vector<string_type> ret;
344
345 for (size_type i = 1; i <= this->Height; ++i)
346 {
347 string_type str;
348 std::getline(stream, str);
349
350 if (str.back() == '\n' || str.back() == '\r')
351 {
352 str.pop_back();
353 }
354
355 str.pop_back();
356 ret.push_back(str);
357
358 if (i == this->Height)
359 {
360 ret.back().pop_back();
361 }
362 }
363
364 return ret;

600 public:
601 using string_type = StringType;
602 using char_type = typename string_type::value_type;
603 using traits_type = typename string_type::traits_type;
604 using size_type = typename string_type::size_type;

609 size_type gap;

611 protected:
618 void trim_deep(Figs_type &figs, Figc_type &figc) const
619 {
620 std::vector<size_type> elem;
621
622 for (size_type i = 0; i < figs.size(); ++i)
623 {
624 int lcount = 0, rcount = 0;
625
626 for (auto itr = figs[i].rbegin(); itr != figs[i].rend(); ++itr)
627 {
628 if (*itr == ' ')
629 ++lcount;
630 else
631 break;
632 }
633
634 for (auto itr = figc[i].begin(); itr != figc[i].end(); ++itr)
635 {
636 if (*itr == ' ')
637 ++rcount;
638 else
639 break;
640 }
641
642 elem.push_back(lcount + rcount);
643 }
644
645 size_type space = *std::min_element(elem.begin(), elem.end());
646
647 for (size_type i = 0; i < figs.size(); ++i)
648 {
649 size_type siz = space;
650
651 while (siz > 0 && figs[i].back() == ' ')
652 {
653 figs[i].pop_back();
654 --siz;
655 }
656
657 figc[i].erase(0, siz);
658 }
659 }

732 template <class StringType>
733 class BasicSmushed : public BasicKerning<StringType>
734 {
735 public:
736 using string_type = StringType;
737 using char_type = typename string_type::value_type;
738 using traits_type = typename string_type::traits_type;
739 using size_type = typename string_type::size_type;
740 using Figc_type = std::vector<string_type>;
741 using Figs_type = std::vector<string_type>;
742
743 private:
750 char_type smush_rules(char_type lc, char_type rc) const
751 {
752 //()
753 if (lc == ' ')
754 {
755 return rc;
756 }
757
758 if (rc == ' ')
759 {
760 return lc;
761 }
762
763 //(Equal character smush)
764 if (lc == rc)
765 {
766 return rc;
767 }
768
769 //(Underscores smush)
770 if (lc == '_' && this->cvt("|/\\[]{}()<>").find(rc) != string_type::npos)
771 {
772 return rc;
773 }
774
775 if (rc == '_' && this->cvt("|/\\[]{}()<>").find(lc) != string_type::npos)
776 {
777 return lc;
778 }
779
780 //(Hierarchy Smushing)
781 auto find_class = [](char_type ch) -> size_type
782 {
783 if (ch == '|')
784 {
785 return 1;
786 }
787
788 if (ch == '/' || ch == '\\')
789 {
790 return 3;
791 }
792
793 if (ch == '[' || ch == ']')
794 {
795 return 4;
796 }
797
798 if (ch == '{' || ch == '}')
799 {
800 return 5;
801 }
802
803 if (ch == '(' || ch == ')')
804 {
805 return 6;
806 }
807
808 return 0;
809 };
810
811 size_type c_lc = find_class(lc);
812 size_type c_rc = find_class(rc);
813
814 if (c_lc > c_rc)
815 {
816 return lc;
817 }
818
819 if (c_rc > c_lc)
820 {
821 return rc;
822 }
823
824 //(Opposite smush)
825 if (lc == '[' && rc == ']')
826 {
827 return '|';
828 }
829
830 if (lc == ']' && rc == '[')
831 {
832 return '|';
833 }
834
835 if (lc == '{' && rc == '}')
836 {
837 return '|';
838 }
839
840 if (lc == '}' && rc == '{')
841 {
842 return '|';
843 }
844
845 if (lc == '(' && rc == ')')
846 {
847 return '|';
848 }
849
850 if (lc == ')' && rc == '(')
851 {
852 return '|';
853 }
854
855 //(Big X smush)
856 if (lc == '/' && rc == '\\')
857 {
858 return '|';
859 }
860
861 if (lc == '\\' && rc == '/')
862 {
863 return 'Y';
864 }
865
866 if (lc == '>' && rc == '<')
867 {
868 return 'X';
869 }
870
871 //(universel smush)
872 return lc;
873 }

881 void smush(Figs_type &figs, Figc_type figc, char_type hb) const
882 {
883 bool smushble = true;
884
885 for (size_type i = 0; i < figs.size(); ++i)
886 {
887 if (figs[i].size() == 0 || figc[i].size() == 0)
888 {
889 smushble = false;
890 }
891 else if ((figs[i].back() == hb) && !(figc[i].front() == hb))
892 {
893 smushble = false;
894 }
895 }
896
897 if (smushble)
898 {
899 for (size_type i = 0; i < figs.size(); ++i)
900 {
901 char_type val = smush_rules(figs[i].back(), figc[i].front());
902 figs[i].pop_back();
903 figc[i].erase(0, 1);
904 figs[i] += string_type(1, val) + figc[i];
905 }
906 }
907 else
908 {
909 for (size_type i = 0; i < figs.size(); ++i)
910 {
911 figs[i] += figc[i];
912 }
913 }
914 }
