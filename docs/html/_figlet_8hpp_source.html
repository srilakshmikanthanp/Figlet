753  {
754  //()
755  if (lc == ' ')
756  {
757  return rc;
758  }
759 
760  if (rc == ' ')
761  {
762  return lc;
763  }
764 
765  //(Equal character smush)
766  if (lc == rc)

768  return rc;
769  }
770

772  if (lc == '_' && this->cvt("|/\\[]{}()<>").find(rc) != string_type::npos)
773  {
774  return rc;
775  }
776 
777  if (rc == '_' && this->cvt("|/\\[]{}()<>").find(lc) != string_type::npos)
778  {
779  return lc;
780  }
781

783  auto find_class = [](char_type ch) -> size_type {
784  if (ch == '|')
785  {
786  return 1;
787  }
788 
789  if (ch == '/' || ch == '\\')
790  {
791  return 3;
792  }
793 
794  if (ch == '[' || ch == ']')
795  {
796  return 4;
797  }
798 
799  if (ch == '{' || ch == '}')
800  {
801  return 5;
802  }
803 
804  if (ch == '(' || ch == ')')
805  {
806  return 6;
807  }

813  size_type c_rc = find_class(rc);
814 
815  if (c_lc > c_rc)
816  {
817  return lc;
818  }
819 
820  if (c_rc > c_lc)
821  {
822  return rc;
823  }

827  {

839  }

844  }

849  }
