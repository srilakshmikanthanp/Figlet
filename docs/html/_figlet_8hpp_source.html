600  {
601  std::vector<size_type> elem;
602 
603  for (size_type i = 0; i < figs.size(); ++i)
604  {

620  break;
621  }
622 
623  elem.push_back(lcount + rcount);
624  }
625

627 
628  for (size_type i = 0; i < figs.size(); ++i)
629  {
630  size_type siz = space;
631 
632  while (siz > 0 && figs[i].back() == ' ')
633  {
634  figs[i].pop_back();

636  }
637 
638  figc[i].erase(0, siz);
639  }

739  char_type smush_rules(char_type lc, char_type rc) const
740  {
741  //()
742  if (lc == ' ')
743  {
744  return rc;
745  }
746 
747  if (rc == ' ')
748  {
749  return lc;
750  }
751 
752  //(Equal character smush)
753  if (lc == rc)
754  {
755  return rc;
756  }
757 
758  //(Underscores smush)
759  if (lc == '_' && this->cvt("|/\\[]{}()<>").find(rc) != string_type::npos)
760  {
761  return rc;
762  }
763 
764  if (rc == '_' && this->cvt("|/\\[]{}()<>").find(lc) != string_type::npos)
765  {
766  return lc;
767  }
768 
769  //(Hierarchy Smushing)
770  auto find_class = [](char_type ch) -> size_type {
771  if (ch == '|')
772  {
773  return 1;
774  }
775 
776  if (ch == '/' || ch == '\\')
777  {
778  return 3;
779  }
780 
781  if (ch == '[' || ch == ']')
782  {
783  return 4;
784  }
785 
786  if (ch == '{' || ch == '}')
787  {
788  return 5;
789  }
790 
791  if (ch == '(' || ch == ')')
792  {
793  return 6;
794  }
795 
796  return 0;
797  };
798

800  size_type c_rc = find_class(rc);
801 
802  if (c_lc > c_rc)
803  {
804  return lc;
805  }
806 
807  if (c_rc > c_lc)
808  {
809  return rc;
810  }
811 
812  //(Opposite smush)
813  if (lc == '[' && rc == ']')
814  {
815  return '|';
816  }
817 
818  if (lc == ']' && rc == '[')
819  {
820  return '|';
821  }
822 
823  if (lc == '{' && rc == '}')
824  {
825  return '|';
826  }
827 
828  if (lc == '}' && rc == '{')
829  {
830  return '|';
831  }
832 
833  if (lc == '(' && rc == ')')
834  {
835  return '|';
836  }
837 
838  if (lc == ')' && rc == '(')
839  {
840  return '|';
841  }
842

844  if (lc == '/' && rc == '\\')
845  {

847  }
848
